if(MSVC)
	FILE(GLOB LIBRARY_FILES_C "*.cpp" "*.rc")
else()
	FILE(GLOB LIBRARY_FILES_C "*.cpp")
endif()

target_link_libraries(Tests Common IO)  # Add tinyxml2 if required

	RUNTIME DESTINATION "${INSTALL_BIN_DIR}" COMPONENT bin)
